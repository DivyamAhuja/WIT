Javascript Loops



                for loop:
                    syntax:
                        for(initialization statement; condition; update statement){
                            ...loop body
                        }
            


                while loop:
                    syntax:
                        initialization statement;
                        while(condition){
                            ...loop body;
                            update statement;
                        }
            


                while loop:
                    syntax:
                        initialization statement;
                        do{
                            ...loop body;
                            update statement;
                        }while(condition);
